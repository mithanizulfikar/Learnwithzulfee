function doPost(e) {
  const data = JSON.parse(e.postData.contents);

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  sheet.appendRow([
    new Date(),
    data.FullName,
    data.FatherName,
    data.Gender,
    data.DOB,
    data.CNIC,
    data.Phone,
    data.Email,
    data.Address,
    data.Qualification,
    data.Course,
    data.BatchTiming,
    data.Mode,
    data.Source
  ]);

MailApp.sendEmail({
    to: data.Email,
    subject: "Registration Confirmation",
    body: "Thank you " + data.FullName
  });

  return ContentService
    .createTextOutput("success")
    .setMimeType(ContentService.MimeType.TEXT);
}
